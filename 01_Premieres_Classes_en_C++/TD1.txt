Allocation statique:
	{
		int a;
		int tab[10];
	}
	
Allocation dynamique:
	{
		char* ch;
		ch = new char[strlen("truc")+1];
		strcpy(ch, "truc");
	}
	
Important to restore memory after usage using:
	delete[]ch;
